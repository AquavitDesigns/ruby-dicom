119:     def execute(verbose=false)
120:       # Search through the folders to gather all the files to be anonymized:

121:       puts "*******************************************************"
122:       puts "Initiating anonymization process."
123:       start_time = Time.now.to_f
124:       puts "Searching for files..."
125:       load_files
126:       puts "Done."
127:       if @files.length > 0
128:         if @tags.length > 0
129:           puts @files.length.to_s + " files have been identified in the specified folder(s)."
130:           if @write_path
131:             # Determine the write paths, as anonymized files will be written to a separate location:

132:             puts "Processing write paths..."
133:             process_write_paths
134:             puts "Done"
135:           else
136:             # Overwriting old files:

137:             puts "Separate write folder not specified. Will overwrite existing DICOM files."
138:             @write_paths = @files
139:           end
140:           # If the user wants enumeration, we need to prepare variables for storing

141:           # existing information associated with each tag:

142:           create_enum_hash if @enumeration
143:           # Start the read/update/write process:

144:           puts "Initiating read/update/write process (This may take some time)..."
145:           # Monitor whether every file read/write was successful:

146:           all_read = true
147:           all_write = true
148:           files_written = 0
149:           files_failed_read = 0
150:           @files.each_index do |i|
151:             # Read existing file to DICOM object:

152:             obj = DICOM::DObject.new(@files[i], :verbose => verbose)
153:             if obj.read_success
154:               # Anonymize the desired tags:

155:               @tags.each_index do |j|
156:                 positions = obj.get_pos(@tags[j])
157:                 positions.each do |pos|
158:                   if @blank
159:                     value = ""
160:                   elsif @enumeration
161:                     old_value = obj.get_value(pos, :silent => true)
162:                     # Only launch enumeration logic if tag exists:

163:                     if old_value
164:                       value = get_enumeration_value(old_value, j)
165:                     else
166:                       value = ""
167:                     end
168:                   else
169:                     # Value is simply value in array:

170:                     value = @values[j]
171:                   end
172:                   # Update DICOM object with new value:

173:                   obj.set_value(value, pos, :create => false, :silent => true)
174:                 end
175:               end
176:               # Remove private tags?

177:               obj.remove_private if @remove_private
178:               # Write DICOM file:

179:               obj.write(@write_paths[i])
180:               if obj.write_success
181:                 files_written += 1
182:               else
183:                 all_write = false
184:               end
185:             else
186:               all_read = false
187:               files_failed_read += 1
188:             end
189:           end
190:           # Finished anonymizing files. Print elapsed time and status of anonymization:

191:           end_time = Time.now.to_f
192:           puts "Anonymization process completed!"
193:           if all_read
194:             puts "All files in specified folder(s) were SUCCESSFULLY read to DICOM objects."
195:           else
196:             puts "Some files were NOT successfully read (#{files_failed_read} files). If folder(s) contain non-DICOM files, this is probably the reason."
197:           end
198:           if all_write
199:             puts "All DICOM objects were SUCCESSFULLY written as DICOM files (#{files_written} files)."
200:           else
201:             puts "Some DICOM objects were NOT succesfully written to file. You are advised to have a closer look (#{files_written} files succesfully written)."
202:           end
203:           # Has user requested enumeration and specified an identity file in which to store the anonymized values?

204:           if @enumeration and @identity_file
205:             puts "Writing identity file."
206:             write_identity_file
207:             puts "Done"
208:           end
209:           elapsed = (end_time-start_time).to_s
210:           puts "Elapsed time: " + elapsed[0..elapsed.index(".")+1] + " seconds"
211:         else
212:           puts "No tags have been selected for anonymization. Aborting."
213:         end
214:       else
215:         puts "No files were found in specified folders. Aborting."
216:       end
217:       puts "*******************************************************"
218:     end
